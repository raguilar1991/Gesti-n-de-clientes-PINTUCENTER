    <script>
    /* ---------- abrir modal y cargar productos ---------- */

    async function abrirModalProductos(event) {
        if (event) event.preventDefault(); // ‚úÖ Evita que el formulario se recargue
  
        const modal = new bootstrap.Modal(document.getElementById('modalProductos'));
        const tablaBody = document.getElementById('tablaProductosBody');
        tablaBody.innerHTML = '<tr><td colspan="6" class="text-center">Cargando...</td></tr>';
  
        const response = await fetch('/productos_json');
        productos = await response.json();

        renderizarProductos(productos);
        modal.show();

        document.getElementById('buscarProducto').addEventListener('input', e => {
            const q = e.target.value.toLowerCase();
            const filtrados = productos.filter(p =>
                p['Nombre producto'].toLowerCase().includes(q) ||
                p['Marca'].toLowerCase().includes(q)
            );
            renderizarProductos(filtrados);
        });
    }

    let productosCache = []; // guardamos productos para filtrar en cliente
    const productosModal = new bootstrap.Modal(document.getElementById('productosModal'), {keyboard:true});

    function renderTablaModal(lista) {
        const tbody = document.querySelector('#tablaModalProductos tbody');
        tbody.innerHTML = '';
        lista.forEach(p => {
            const tr = document.createElement('tr');
            tr.innerHTML = `
                <td>${escapeHtml(p.nombre_producto)}</td>
                <td>${escapeHtml(p.marca || '')}</td>
                <td class="text-end">${Number(p.precio).toFixed(2)}</td>
                <td class="text-center">${p.stock ?? ''}</td>
                <td class="text-center">
                    <button class="btn btn-sm btn-primary" onclick='seleccionarProducto(${JSON.stringify(p).replace(/'/g,"&#39;")})'>Agregar</button>
                </td>
            `;
            // doble clic r√°pido
            tr.addEventListener('dblclick', () => seleccionarProducto(p));
            tbody.appendChild(tr);
        });
    }

    /* filtro de b√∫squeda dentro del modal */
    function filtrarProductos() {
        const q = document.getElementById('buscarProducto').value.trim().toLowerCase();
        const filtrados = productosCache.filter(p => {
            return (p.nombre_producto && p.nombre_producto.toLowerCase().includes(q)) ||
                    (p.marca && p.marca.toLowerCase().includes(q)) ||
                    String(p.id).includes(q);
        });
        renderTablaModal(filtrados);
    }

    /* escape b√°sico para evitar inyecci√≥n al insertar en DOM (nombre/ marca) */
    function escapeHtml(unsafe) {
        if (!unsafe && unsafe !== 0) return '';
        return String(unsafe)
            .replace(/&/g, "&amp;")
            .replace(/</g, "&lt;")
            .replace(/>/g, "&gt;")
            .replace(/"/g, "&quot;")
            .replace(/'/g, "&#039;");
    }

    /* ---------- cuando el usuario elige un producto ---------- */
    function seleccionarProducto(p) {
        // p es objeto con: id, nombre_producto, marca, precio, stock
        productosModal.hide();

        // Intento reusar la fila seleccionada si existe (opcional).
        // Por simplicidad a√±adiremos una nueva fila con los datos del producto.
        addProductToTable(p);
        calcularTotal(); // recalcula totales
    }

    /* a√±ade producto como nueva fila en la tabla de items */
    function addProductToTable(producto) {
        const tbody = document.querySelector('#tablaItems tbody');
        const index = tbody.rows.length + 1;
        const tr = document.createElement('tr');

        tr.innerHTML = `
            <td class="text-center">${index}</td>
            <td>
                <div><strong>${escapeHtml(producto.nombre_producto)}</strong></div>
                <div class="text-muted small">${escapeHtml(producto.marca || '')}</div>
                <input type="hidden" name="producto_id[]" value="${producto.id}">
                <input type="hidden" name="producto_nombre[]" value="${escapeHtml(producto.nombre_producto)}">
            </td>
            <td><input type="number" name="cantidad[]" class="form-control" value="1" min="1" onchange="calcularTotal()"></td>
            <td><input type="number" name="valor_unitario[]" class="form-control" value="${Number(producto.precio).toFixed(2)}" step="0.01" onchange="calcularTotal()"></td>
            <td><input type="number" name="total[]" class="form-control" value="${Number(producto.precio).toFixed(2)}" readonly></td>
            <td class="text-center"><button type="button" class="btn btn-danger btn-compact" onclick="eliminarFila(this)">-</button></td>
        `;
        tbody.appendChild(tr);
    }
    </script>


    <!-- TOGGLE SIDEBAR Y SUBMEN√ö -->
    <script>
    document.addEventListener('DOMContentLoaded', function() { 
        const sidebarToggle = document.getElementById('sidebarToggle'); 
        const wrapper = document.getElementById('wrapper');
        if (sidebarToggle && wrapper) {
            sidebarToggle.addEventListener('click', () => wrapper.classList.toggle('toggled')); 
        }

        const clientesToggle = document.getElementById('clientesToggle'); 
        const submenuItems = document.querySelector('.submenu-items'); 
        const arrow = clientesToggle ? clientesToggle.querySelector('.arrow') : null; 

        if (clientesToggle && submenuItems) { 
            let hideTimeout;

            // M√ìVIL: click para abrir
            clientesToggle.addEventListener('click', function(e) {
                if (window.innerWidth <= 991) {
                    e.preventDefault();
                    const opened = clientesToggle.classList.toggle('open');
                    submenuItems.classList.toggle('show', opened);
                    clientesToggle.setAttribute('aria-expanded', opened ? 'true' : 'false');
                }
            });

            // ESCRITORIO: hover para abrir
            clientesToggle.addEventListener('mouseenter', function() {
                clearTimeout(hideTimeout);
                submenuItems.classList.add('show');
                if (arrow) arrow.style.transform = 'rotate(180deg)';
            });
            clientesToggle.addEventListener('mouseleave', function() {
                hideTimeout = setTimeout(() => {
                    if (window.innerWidth > 991) {
                        submenuItems.classList.remove('show');
                        if (arrow) arrow.style.transform = '';
                    }
                }, 200);
            });

            submenuItems.addEventListener('mouseenter', function() {
                clearTimeout(hideTimeout);
                submenuItems.classList.add('show');
                if (arrow) arrow.style.transform = 'rotate(180deg)';
            });
            submenuItems.addEventListener('mouseleave', function() {
                hideTimeout = setTimeout(() => {
                    if (window.innerWidth > 991) {
                        submenuItems.classList.remove('show');
                        if (arrow) arrow.style.transform = '';
                    }
                }, 200);
            });

            window.addEventListener('resize', function() {
                if (window.innerWidth > 991) {
                    clientesToggle.classList.remove('open');
                    clientesToggle.setAttribute('aria-expanded', 'false');
                    submenuItems.classList.remove('show');
                    if (arrow) arrow.style.transform = '';
                }
            });
        }
    });
    </script>

    <script>
    // Rellenar correo/telefono al seleccionar cliente (usa los data-atributos de <option>)
    document.getElementById('clienteSelect').addEventListener('change', function() {
        const opt = this.options[this.selectedIndex];
        document.getElementById('correo').value = opt.dataset.correo || '';
        document.getElementById('telefono').value = opt.dataset.telefono || '';
    });

    // Calcular totales
    function calcularTotal() {
        const tbody = document.querySelector('#tablaItems tbody');
        let subtotal = 0;
        for (let r = 0; r < tbody.rows.length; r++) {
            const row = tbody.rows[r];
            const cant = parseFloat(row.querySelector('input[name="cantidad[]"]').value) || 0;
            const valor = parseFloat(row.querySelector('input[name="valor_unitario[]"]').value) || 0;
            const total = cant * valor;
            const totalInput = row.querySelector('input[name="total[]"]');
            totalInput.value = total.toFixed(2);
            subtotal += total;
            // actualizar √≠ndice del item
            row.cells[0].textContent = r + 1;
        }
        document.getElementById('subtotal').value = subtotal.toFixed(2);
        const iva = subtotal * 0.15;
        document.getElementById('iva').value = iva.toFixed(2);
        document.getElementById('total_final').value = (subtotal + iva).toFixed(2);
    }

    // Agregar fila
    function agregarFila() {
        const tbody = document.querySelector('#tablaItems tbody');
        const index = tbody.rows.length + 1;
        const tr = document.createElement('tr');
        tr.innerHTML = `
            <td class="text-center">${index}</td>
            <td><input type="text" name="descripcion[]" class="form-control" required></td>
            <td><input type="number" name="cantidad[]" class="form-control" value="1" min="1" onchange="calcularTotal()"></td>
            <td><input type="number" name="valor_unitario[]" class="form-control" value="0.00" step="0.01" onchange="calcularTotal()"></td>
            <td><input type="number" name="total[]" class="form-control" value="0.00" readonly></td>
            <td class="text-center"><button type="button" class="btn btn-danger btn-compact" onclick="eliminarFila(this)">-</button></td>
        `;
        tbody.appendChild(tr);
        calcularTotal();
        // focus en el input de descripci√≥n reci√©n creado
        tr.querySelector('input[name="descripcion[]"]').focus();
    }

    // Eliminar fila
    function eliminarFila(btn) {
        const row = btn.closest('tr');
        row.remove();
        calcularTotal();
    }

    // Ejecutar c√°lculo inicial
    document.addEventListener('DOMContentLoaded', function() {
        calcularTotal();
    });
    </script>
